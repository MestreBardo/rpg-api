/users/login:
    post:
      tags: 
        - users
      summary: Efetua o login na API
      operationId: Login
      description: |
        Endpoint utilizado para efetuar login e acessar outros metodos.
      requestBody:
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/Login'
      responses:
        "200":
          description: Login realizado com sucesso.
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Authorization'
        "401":
          description: Não foi possivel efetuar o login.
          content:
            application/json:
              schema:
                type: array
                items:
                  type: string
                  example: Email or Password is not right
  /users:
    get:
      security: 
        - token: []
      tags:
        - users
      summary: Retorna lista com usúarios
      operationId: Retrieve Users
      description: |
        Retorna uma lista de usuários ativos nos sistema, podem ser utilizados  
        parametros para uma busca mais refinada.
      responses:
        '200':
          description: Resultado da busca atendendo aos requisitos
          content:
            application/json:
              schema:
                type: array
                items: 
                  $ref: '#/components/schemas/User'
        '401': 
          description: Não autorizado. Token invalido ou não submetido
          content:
            application/json:
              schema:
                type: array
                items:
                  type: string
                  example: Unauthorized
        '500':
          description: Erro interno de servidor
          content:
            application/json:
              schema:
                type: array
                items:
                  type: string
                  example: Error Message of Server
    post:
      tags:
        - users
      summary: Criar usuário no sistema
      operationId: Create User
      description: |
        Permite a criação de usuário na plataforma.
      requestBody:
        content:
          application/json:
            schema:
              allOf: 
                - $ref: '#/components/schemas/User'
                - type: object
                  required:
                    - name
                    - surname
                    - username
                    - password
                    - email
                    - birthday
                    - country
                    - gender
      responses: 
        '200':
          description: Usuário criado com sucesso.
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Authorization'
  /users/{id}:
    parameters: 
      - name: id
        in: path
        description: ID do usuário que consta no banco
        required: true
        schema:
          type: string
    get:
      security: 
        - token: []
      tags: 
        - users
      summary: Retorna o úsuario pesquisado
      operationId: Retrieve User
      description: |
        Retorna o usuário requisitado se o mesmo estiver ativo.
      responses:
        '200':
          description: Resultado da busca. Usúario encontrado e ativo.
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/User'
        '404':
          description: Usuário não encontrado
          content:
            application/json:
              schema:
                type: array
                items:
                  type: string
                  example: This user is inactivated
    patch: 
      security:
        - token: []
      tags: 
        - users
      summary: Realiza a atualização de dados enviados
      description: |
        Realiza a atualização dos dados do usuário. Enviar somente os dados que deseja atualizar
      requestBody: 
        content: 
          application/json:
           schema: 
            $ref: '#/components/schemas/User'
      responses: 
        "200":
          description: Usuário Atualizado com sucesso.
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/User'
        "401":
          description: Sem permissão para este usuário
          content:
            application/json:
              schema:
                type: array
                items:
                  type: string
                  example: You don't owner this user
        "404":
          description: Usuário não encontrado
          content:
            application/json:
              schema:
                allOf: 
                  - $ref: '#/components/schemas/Error'
                properties: 
                  resource:
                    example: USERS
                  messages:
                    example: 
                     - User don't exists in database
        "410":
          description: Usuário inativado
          content:
            application/json:
              schema:
                allOf: 
                  - $ref: '#/components/schemas/Error'
                properties: 
                  resource:
                    example: USERS
                  messages:
                    example: 
                     - User is not active
